package pages;

import Utilites.BrowserActions;
import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import sun.jvm.hotspot.utilities.Assert;

public class LoginPage {
     WebDriver driver;
     WebDriver driver = driver;
    //Locators
    By username_textbox = By.name("username");
    By passaword_textbox = By.name("password");
    By login_button = By.className("oxd-button oxd-button--medium oxd-button--main orangehrm-login-button");

    //Actions
    public void navigatetologinpage()
    {
        driver.get("https://opensource-demo.orangehrmlive.com/web/index.php/auth/login");
    }

    public void enterusername()
    {
        driver.findElement(username_textbox).sendKeys("tomsmith");
    }

    public void enterpassword()
    {
        driver.findElement(passaword_textbox).sendKeys("SuperSecretPassword!");
    }

    public void clickloginbutton()
    {
        driver.findElement(login_button).click();
    }

    //Assertaions
    public void validateSuccessfullogin() {
        String alertText = BrowserActions.getAlertText(driver);
        Assert.assertEquals(alertText, "Sign in successful.");
        BrowserActions.acceptAlerts(driver);

    }

}
